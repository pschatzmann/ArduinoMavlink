74/**
75 * @brief create a signature block for a packet
76 */
77MAVLINK_HELPER uint8_t mavlink_sign_packet(mavlink_signing_t *signing,
78 uint8_t signature[MAVLINK_SIGNATURE_BLOCK_LEN],
79 const uint8_t *header, uint8_t header_len,
80 const uint8_t *packet, uint8_t packet_len,
81 const uint8_t crc[2])
82{
83 mavlink_sha256_ctx ctx;
84 union {
85 uint64_t t64;
86 uint8_t t8[8];
87 } tstamp;
88 if (signing == NULL || !(signing->flags & MAVLINK_SIGNING_FLAG_SIGN_OUTGOING)) {
89 return 0;
90 }
91 signature[0] = signing->link_id;
92 tstamp.t64 = signing->timestamp;
93 memcpy(&signature[1], tstamp.t8, 6);
94 signing->timestamp++;
95
96 mavlink_sha256_init(&ctx);
97 mavlink_sha256_update(&ctx, signing->secret_key, sizeof(signing->secret_key));
98 mavlink_sha256_update(&ctx, header, header_len);
99 mavlink_sha256_update(&ctx, packet, packet_len);
100 mavlink_sha256_update(&ctx, crc, 2);
101 mavlink_sha256_update(&ctx, signature, 7);
102 mavlink_sha256_final_48(&ctx, &signature[7]);
103
104 return MAVLINK_SIGNATURE_BLOCK_LEN;
105}
106#endif

111 * @param payload Serialised payload buffer.
112 * @param length Length of full-width payload buffer.
113 * @return Length of payload after zero-filled bytes are trimmed.
114 */
115MAVLINK_HELPER uint8_t _mav_trim_payload(const char *payload, uint8_t length)
116{
117 while (length > 1 && payload[length-1] == 0) {
118 length--;
119 }

123#ifndef MAVLINK_NO_SIGNATURE_CHECK
124/**
125 * @brief check a signature block for a packet
126 */
127MAVLINK_HELPER bool mavlink_signature_check(mavlink_signing_t *signing,
128 mavlink_signing_streams_t *signing_streams,
129 const mavlink_message_t *msg)
130{
131 if (signing == NULL) {
132 return true;
133 }
134 const uint8_t *p = (const uint8_t *)&msg->magic;
135 const uint8_t *psig = msg->signature;
136 const uint8_t *incoming_signature = psig+7;
137 mavlink_sha256_ctx ctx;
138 uint8_t signature[6];
139 uint16_t i;
140
141 mavlink_sha256_init(&ctx);
142 mavlink_sha256_update(&ctx, signing->secret_key, sizeof(signing->secret_key));
143 mavlink_sha256_update(&ctx, p, MAVLINK_NUM_HEADER_BYTES);
144 mavlink_sha256_update(&ctx, _MAV_PAYLOAD(msg), msg->len);
145 mavlink_sha256_update(&ctx, msg->ck, 2);
146 mavlink_sha256_update(&ctx, psig, 1+6);
147 mavlink_sha256_final_48(&ctx, signature);
148 if (memcmp(signature, incoming_signature, 6) != 0) {
149 signing->last_status = MAVLINK_SIGNING_STATUS_BAD_SIGNATURE;
150 return false;
151 }
152
153 // now check timestamp
154 union tstamp {
155 uint64_t t64;
156 uint8_t t8[8];
157 } tstamp;
158 uint8_t link_id = psig[0];
159 tstamp.t64 = 0;
160 memcpy(tstamp.t8, psig+1, 6);
161
162 if (signing_streams == NULL) {
163 signing->last_status = MAVLINK_SIGNING_STATUS_NO_STREAMS;
164 return false;
165 }
166
167 // find stream
168 for (i=0; i<signing_streams->num_signing_streams; i++) {
169 if (msg->sysid == signing_streams->stream[i].sysid &&
170 msg->compid == signing_streams->stream[i].compid &&
171 link_id == signing_streams->stream[i].link_id) {
172 break;
173 }
174 }
175 if (i == signing_streams->num_signing_streams) {
176 if (signing_streams->num_signing_streams >= MAVLINK_MAX_SIGNING_STREAMS) {
177 // over max number of streams
178 signing->last_status = MAVLINK_SIGNING_STATUS_TOO_MANY_STREAMS;
179 return false;
180 }
181 // new stream. Only accept if timestamp is not more than 1 minute old by default
182 if (tstamp.t64 + MAVLINK_SIGNING_TIMESTAMP_LIMIT*100000UL < signing->timestamp) {
183 signing->last_status = MAVLINK_SIGNING_STATUS_OLD_TIMESTAMP;
184 return false;
185 }
186 // add new stream
187 signing_streams->stream[i].sysid = msg->sysid;
188 signing_streams->stream[i].compid = msg->compid;
189 signing_streams->stream[i].link_id = link_id;
190 signing_streams->num_signing_streams++;
191 } else {
192 union tstamp last_tstamp;
193 last_tstamp.t64 = 0;
194 memcpy(last_tstamp.t8, signing_streams->stream[i].timestamp_bytes, 6);
195 if (tstamp.t64 <= last_tstamp.t64) {
196 // repeating old timestamp
197 signing->last_status = MAVLINK_SIGNING_STATUS_REPLAY;
198 return false;
199 }
200 }
201
202 // remember last timestamp
203 memcpy(signing_streams->stream[i].timestamp_bytes, psig+1, 6);
204
205 // our next timestamp must be at least this timestamp
206 if (tstamp.t64 > signing->timestamp) {
207 signing->timestamp = tstamp.t64;
208 }
209 signing->last_status = MAVLINK_SIGNING_STATUS_OK;
210 return true;
211}
212#endif

215/**
216 * @brief Finalize a MAVLink message with channel assignment
217 *
218 * This function calculates the checksum and sets length and aircraft id correctly.
219 * It assumes that the message id and the payload are already correctly set. This function
220 * can also be used if the message header has already been written before (as in mavlink_msg_xxx_pack
221 * instead of mavlink_msg_xxx_pack_headerless), it just introduces little extra overhead.
222 *
223 * @param msg Message to finalize
224 * @param system_id Id of the sending (this) system, 1-127
225 * @param length Message length
226 */
227MAVLINK_HELPER uint16_t mavlink_finalize_message_buffer(mavlink_message_t* msg, uint8_t system_id, uint8_t component_id,
228 mavlink_status_t* status, uint8_t min_length, uint8_t length, uint8_t crc_extra)
229{
230 bool mavlink1 = (status->flags & MAVLINK_STATUS_FLAG_OUT_MAVLINK1) != 0;
231#ifndef MAVLINK_NO_SIGN_PACKET
232 bool signing = (!mavlink1) && status->signing && (status->signing->flags & MAVLINK_SIGNING_FLAG_SIGN_OUTGOING);
233#else
234 bool signing = false;
235#endif
236 uint8_t signature_len = signing? MAVLINK_SIGNATURE_BLOCK_LEN : 0;
237 uint8_t header_len = MAVLINK_CORE_HEADER_LEN+1;
238 uint8_t buf[MAVLINK_CORE_HEADER_LEN+1];
239 if (mavlink1) {
240 msg->magic = MAVLINK_STX_MAVLINK1;
241 header_len = MAVLINK_CORE_HEADER_MAVLINK1_LEN+1;
242 } else {
243 msg->magic = MAVLINK_STX;
244 }
245 msg->len = mavlink1?min_length:_mav_trim_payload(_MAV_PAYLOAD(msg), length);
246 msg->sysid = system_id;
247 msg->compid = component_id;
248 msg->incompat_flags = 0;
249 if (signing) {
250 msg->incompat_flags |= MAVLINK_IFLAG_SIGNED;
251 }
252 msg->compat_flags = 0;
253 msg->seq = status->current_tx_seq;
254 status->current_tx_seq = status->current_tx_seq + 1;
255
256 // form the header as a byte array for the crc
257 buf[0] = msg->magic;
258 buf[1] = msg->len;
259 if (mavlink1) {
260 buf[2] = msg->seq;
261 buf[3] = msg->sysid;
262 buf[4] = msg->compid;
263 buf[5] = msg->msgid & 0xFF;
264 } else {
265 buf[2] = msg->incompat_flags;
266 buf[3] = msg->compat_flags;
267 buf[4] = msg->seq;
268 buf[5] = msg->sysid;
269 buf[6] = msg->compid;
270 buf[7] = msg->msgid & 0xFF;
271 buf[8] = (msg->msgid >> 8) & 0xFF;
272 buf[9] = (msg->msgid >> 16) & 0xFF;
273 }
274
275 uint16_t checksum = crc_calculate(&buf[1], header_len-1);
276 crc_accumulate_buffer(&checksum, _MAV_PAYLOAD(msg), msg->len);
277 crc_accumulate(crc_extra, &checksum);
278 mavlink_ck_a(msg) = (uint8_t)(checksum & 0xFF);
279 mavlink_ck_b(msg) = (uint8_t)(checksum >> 8);
280
281 msg->checksum = checksum;
282
283#ifndef MAVLINK_NO_SIGN_PACKET
284 if (signing) {
285 mavlink_sign_packet(status->signing,
286 msg->signature,
287 (const uint8_t *)buf, header_len,
288 (const uint8_t *)_MAV_PAYLOAD(msg), msg->len,
289 (const uint8_t *)_MAV_PAYLOAD(msg)+(uint16_t)msg->len);
290 }
291#endif
292
293 return msg->len + header_len + 2 + signature_len;
294}

320/**
321 * @brief Finalize a MAVLink message with channel assignment and send
322 */
323MAVLINK_HELPER void _mav_finalize_message_chan_send(mavlink_channel_t chan, uint32_t msgid,
324 const char *packet,
325 uint8_t min_length, uint8_t length, uint8_t crc_extra)
326{
327 uint16_t checksum;
328 uint8_t buf[MAVLINK_NUM_HEADER_BYTES];
329 uint8_t ck[2];
330 mavlink_status_t *status = mavlink_get_channel_status(chan);
331 uint8_t header_len = MAVLINK_CORE_HEADER_LEN;
332 uint8_t signature_len = 0;
333 uint8_t signature[MAVLINK_SIGNATURE_BLOCK_LEN];
334 bool mavlink1 = (status->flags & MAVLINK_STATUS_FLAG_OUT_MAVLINK1) != 0;
335 bool signing = (!mavlink1) && status->signing && (status->signing->flags & MAVLINK_SIGNING_FLAG_SIGN_OUTGOING);
336
337 if (mavlink1) {
338 length = min_length;
339 if (msgid > 255) {
340 // can't send 16 bit messages
341 _mav_parse_error(status);
342 return;
343 }
344 header_len = MAVLINK_CORE_HEADER_MAVLINK1_LEN;
345 buf[0] = MAVLINK_STX_MAVLINK1;
346 buf[1] = length;
347 buf[2] = status->current_tx_seq;
348 buf[3] = mavlink_system.sysid;
349 buf[4] = mavlink_system.compid;
350 buf[5] = msgid & 0xFF;
351 } else {
352 uint8_t incompat_flags = 0;
353 if (signing) {
354 incompat_flags |= MAVLINK_IFLAG_SIGNED;
355 }
356 length = _mav_trim_payload(packet, length);
357 buf[0] = MAVLINK_STX;
358 buf[1] = length;
359 buf[2] = incompat_flags;
360 buf[3] = 0; // compat_flags
361 buf[4] = status->current_tx_seq;
362 buf[5] = mavlink_system.sysid;
363 buf[6] = mavlink_system.compid;
364 buf[7] = msgid & 0xFF;
365 buf[8] = (msgid >> 8) & 0xFF;
366 buf[9] = (msgid >> 16) & 0xFF;
367 }
368 status->current_tx_seq++;
369 checksum = crc_calculate((const uint8_t*)&buf[1], header_len);
370 crc_accumulate_buffer(&checksum, packet, length);
371 crc_accumulate(crc_extra, &checksum);
372 ck[0] = (uint8_t)(checksum & 0xFF);
373 ck[1] = (uint8_t)(checksum >> 8);
374
375#ifndef MAVLINK_NO_SIGN_PACKET
376 if (signing) {
377 // possibly add a signature
378 signature_len = mavlink_sign_packet(status->signing, signature, buf, header_len+1,
379 (const uint8_t *)packet, length, ck);
380 }
381#endif
382
383 MAVLINK_START_UART_SEND(chan, header_len + 3 + (uint16_t)length + (uint16_t)signature_len);
384 _mavlink_send_uart(chan, (const char *)buf, header_len+1);
385 _mavlink_send_uart(chan, packet, length);
386 _mavlink_send_uart(chan, (const char *)ck, 2);
387 if (signature_len != 0) {
388 _mavlink_send_uart(chan, (const char *)signature, signature_len);
389 }
390 MAVLINK_END_UART_SEND(chan, header_len + 3 + (uint16_t)length + (uint16_t)signature_len);
391}

393/**
394 * @brief re-send a message over a uart channel
395 * this is more stack efficient than re-marshalling the message
396 * If the message is signed then the original signature is also sent
397 */
398MAVLINK_HELPER void _mavlink_resend_uart(mavlink_channel_t chan, const mavlink_message_t *msg)
399{
400 uint8_t ck[2];
401
402 ck[0] = (uint8_t)(msg->checksum & 0xFF);
403 ck[1] = (uint8_t)(msg->checksum >> 8);
404 // XXX use the right sequence here
405
406 uint8_t header_len;
407 uint8_t signature_len;
408
409 if (msg->magic == MAVLINK_STX_MAVLINK1) {
410 header_len = MAVLINK_CORE_HEADER_MAVLINK1_LEN + 1;
411 signature_len = 0;
412 MAVLINK_START_UART_SEND(chan, header_len + msg->len + 2 + signature_len);
413 // we can't send the structure directly as it has extra mavlink2 elements in it
414 uint8_t buf[MAVLINK_CORE_HEADER_MAVLINK1_LEN + 1];
415 buf[0] = msg->magic;
416 buf[1] = msg->len;
417 buf[2] = msg->seq;
418 buf[3] = msg->sysid;
419 buf[4] = msg->compid;
420 buf[5] = msg->msgid & 0xFF;
421 _mavlink_send_uart(chan, (const char*)buf, header_len);
422 } else {
423 header_len = MAVLINK_CORE_HEADER_LEN + 1;
424 signature_len = (msg->incompat_flags & MAVLINK_IFLAG_SIGNED)?MAVLINK_SIGNATURE_BLOCK_LEN:0;
425 MAVLINK_START_UART_SEND(chan, header_len + msg->len + 2 + signature_len);
426 uint8_t buf[MAVLINK_CORE_HEADER_LEN + 1];
427 buf[0] = msg->magic;
428 buf[1] = msg->len;
429 buf[2] = msg->incompat_flags;
430 buf[3] = msg->compat_flags;
431 buf[4] = msg->seq;
432 buf[5] = msg->sysid;
433 buf[6] = msg->compid;
434 buf[7] = msg->msgid & 0xFF;
435 buf[8] = (msg->msgid >> 8) & 0xFF;
436 buf[9] = (msg->msgid >> 16) & 0xFF;
437 _mavlink_send_uart(chan, (const char *)buf, header_len);

448/**
449 * @brief Pack a message to send it over a serial byte stream
450 */
451MAVLINK_HELPER uint16_t mavlink_msg_to_send_buffer(uint8_t *buf, const mavlink_message_t *msg)
452{
453 uint8_t signature_len, header_len;
454 uint8_t *ck;
455 uint8_t length = msg->len;
456
457 if (msg->magic == MAVLINK_STX_MAVLINK1) {
458 signature_len = 0;
459 header_len = MAVLINK_CORE_HEADER_MAVLINK1_LEN;
460 buf[0] = msg->magic;
461 buf[1] = length;
462 buf[2] = msg->seq;
463 buf[3] = msg->sysid;
464 buf[4] = msg->compid;
465 buf[5] = msg->msgid & 0xFF;
466 memcpy(&buf[6], _MAV_PAYLOAD(msg), msg->len);
467 ck = buf + header_len + 1 + (uint16_t)msg->len;
468 } else {
469 length = _mav_trim_payload(_MAV_PAYLOAD(msg), length);
470 header_len = MAVLINK_CORE_HEADER_LEN;
471 buf[0] = msg->magic;
472 buf[1] = length;
473 buf[2] = msg->incompat_flags;
474 buf[3] = msg->compat_flags;
475 buf[4] = msg->seq;
476 buf[5] = msg->sysid;
477 buf[6] = msg->compid;
478 buf[7] = msg->msgid & 0xFF;
479 buf[8] = (msg->msgid >> 8) & 0xFF;
480 buf[9] = (msg->msgid >> 16) & 0xFF;
481 memcpy(&buf[10], _MAV_PAYLOAD(msg), length);
482 ck = buf + header_len + 1 + (uint16_t)length;
483 signature_len = (msg->incompat_flags & MAVLINK_IFLAG_SIGNED)?MAVLINK_SIGNATURE_BLOCK_LEN:0;
484 }
485 ck[0] = (uint8_t)(msg->checksum & 0xFF);
486 ck[1] = (uint8_t)(msg->checksum >> 8);
487 if (signature_len > 0) {
488 memcpy(&ck[2], msg->signature, signature_len);
489 }
490
491 return header_len + 1 + 2 + (uint16_t)length + (uint16_t)signature_len;
492}

518/*
519 return the crc_entry value for a msgid
520*/
521#ifndef MAVLINK_GET_MSG_ENTRY
522MAVLINK_HELPER const mavlink_msg_entry_t *mavlink_get_msg_entry(uint32_t msgid)
523{
524 static const mavlink_msg_entry_t mavlink_message_crcs[] = MAVLINK_MESSAGE_CRCS;
525 /*
526 use a bisection search to find the right entry. A perfect hash may be better
527 Note that this assumes the table is sorted by msgid
528 */
529 uint32_t low=0, high=sizeof(mavlink_message_crcs)/sizeof(mavlink_message_crcs[0]) - 1;
530 while (low < high) {
531 uint32_t mid = (low+1+high)/2;
532 if (msgid < mavlink_message_crcs[mid].msgid) {
533 high = mid-1;
534 continue;
535 }
536 if (msgid > mavlink_message_crcs[mid].msgid) {
537 low = mid;
538 continue;
539 }
540 low = mid;
541 break;
542 }
543 if (mavlink_message_crcs[low].msgid != msgid) {
544 // msgid is not in the table
545 return NULL;
546 }
547 return &mavlink_message_crcs[low];
548}
549#endif // MAVLINK_GET_MSG_ENTRY

580/**
581 * This is a variant of mavlink_frame_char() but with caller supplied
582 * parsing buffers. It is useful when you want to create a MAVLink
583 * parser in a library that doesn't use any global variables
584 *
585 * @param rxmsg parsing message buffer
586 * @param status parsing status buffer
587 * @param c The char to parse
588 *
589 * @param r_message NULL if no message could be decoded, otherwise the message data
590 * @param r_mavlink_status if a message was decoded, this is filled with the channel's stats
591 * @return 0 if no message could be decoded, 1 on good message and CRC, 2 on bad CRC
592 *
593 */
594MAVLINK_HELPER uint8_t mavlink_frame_char_buffer(mavlink_message_t* rxmsg,
595 mavlink_status_t* status,
596 uint8_t c,
597 mavlink_message_t* r_message,
598 mavlink_status_t* r_mavlink_status)
599{
600
601 status->msg_received = MAVLINK_FRAMING_INCOMPLETE;
602
603 switch (status->parse_state)
604 {
605 case MAVLINK_PARSE_STATE_UNINIT:
606 case MAVLINK_PARSE_STATE_IDLE:
607 if (c == MAVLINK_STX)
608 {
609 status->parse_state = MAVLINK_PARSE_STATE_GOT_STX;
610 rxmsg->len = 0;
611 rxmsg->magic = c;
612 status->flags &= ~MAVLINK_STATUS_FLAG_IN_MAVLINK1;
613 mavlink_start_checksum(rxmsg);
614 } else if (c == MAVLINK_STX_MAVLINK1)
615 {
616 status->parse_state = MAVLINK_PARSE_STATE_GOT_STX;
617 rxmsg->len = 0;
618 rxmsg->magic = c;
619 status->flags |= MAVLINK_STATUS_FLAG_IN_MAVLINK1;
620 mavlink_start_checksum(rxmsg);
621 }
622 break;
623
624 case MAVLINK_PARSE_STATE_GOT_STX:
625 if (status->msg_received
626/* Support shorter buffers than the
627 default maximum packet size */
628#if (MAVLINK_MAX_PAYLOAD_LEN < 255)
629 || c > MAVLINK_MAX_PAYLOAD_LEN
630#endif
631 )
632 {
633 status->buffer_overrun++;
634 _mav_parse_error(status);
635 status->msg_received = MAVLINK_FRAMING_INCOMPLETE;
636 status->parse_state = MAVLINK_PARSE_STATE_IDLE;
637 }
638 else
639 {
640 // NOT counting STX, LENGTH, SEQ, SYSID, COMPID, MSGID, CRC1 and CRC2
641 rxmsg->len = c;
642 status->packet_idx = 0;
643 mavlink_update_checksum(rxmsg, c);
644 if (status->flags & MAVLINK_STATUS_FLAG_IN_MAVLINK1) {
645 rxmsg->incompat_flags = 0;
646 rxmsg->compat_flags = 0;
647 status->parse_state = MAVLINK_PARSE_STATE_GOT_COMPAT_FLAGS;
648 } else {
649 status->parse_state = MAVLINK_PARSE_STATE_GOT_LENGTH;
650 }
651 }
652 break;
653
654 case MAVLINK_PARSE_STATE_GOT_LENGTH:
655 rxmsg->incompat_flags = c;
656 if ((rxmsg->incompat_flags & ~MAVLINK_IFLAG_MASK) != 0) {
657 // message includes an incompatible feature flag
658 _mav_parse_error(status);
659 status->msg_received = MAVLINK_FRAMING_INCOMPLETE;
660 status->parse_state = MAVLINK_PARSE_STATE_IDLE;
661 break;
662 }
663 mavlink_update_checksum(rxmsg, c);
664 status->parse_state = MAVLINK_PARSE_STATE_GOT_INCOMPAT_FLAGS;
665 break;
666
667 case MAVLINK_PARSE_STATE_GOT_INCOMPAT_FLAGS:
668 rxmsg->compat_flags = c;
669 mavlink_update_checksum(rxmsg, c);
670 status->parse_state = MAVLINK_PARSE_STATE_GOT_COMPAT_FLAGS;
671 break;
672
673 case MAVLINK_PARSE_STATE_GOT_COMPAT_FLAGS:
674 rxmsg->seq = c;
675 mavlink_update_checksum(rxmsg, c);
676 status->parse_state = MAVLINK_PARSE_STATE_GOT_SEQ;
677 break;
678
679 case MAVLINK_PARSE_STATE_GOT_SEQ:
680 rxmsg->sysid = c;
681 mavlink_update_checksum(rxmsg, c);
682 status->parse_state = MAVLINK_PARSE_STATE_GOT_SYSID;
683 break;
684
685 case MAVLINK_PARSE_STATE_GOT_SYSID:
686 rxmsg->compid = c;
687 mavlink_update_checksum(rxmsg, c);
688 status->parse_state = MAVLINK_PARSE_STATE_GOT_COMPID;
689 break;
690
691 case MAVLINK_PARSE_STATE_GOT_COMPID:
692 rxmsg->msgid = c;
693 mavlink_update_checksum(rxmsg, c);
694 if (status->flags & MAVLINK_STATUS_FLAG_IN_MAVLINK1) {
695 if(rxmsg->len > 0) {
696 status->parse_state = MAVLINK_PARSE_STATE_GOT_MSGID3;
697 } else {
698 status->parse_state = MAVLINK_PARSE_STATE_GOT_PAYLOAD;
699 }
700#ifdef MAVLINK_CHECK_MESSAGE_LENGTH
701 if (rxmsg->len < mavlink_min_message_length(rxmsg) ||
702 rxmsg->len > mavlink_max_message_length(rxmsg)) {
703 _mav_parse_error(status);
704 status->parse_state = MAVLINK_PARSE_STATE_IDLE;
705 break;
706 }
707#endif
708 } else {
709 status->parse_state = MAVLINK_PARSE_STATE_GOT_MSGID1;
710 }
711 break;
712
713 case MAVLINK_PARSE_STATE_GOT_MSGID1:
714 rxmsg->msgid |= ((uint32_t)c)<<8;
715 mavlink_update_checksum(rxmsg, c);
716 status->parse_state = MAVLINK_PARSE_STATE_GOT_MSGID2;
717 break;
718
719 case MAVLINK_PARSE_STATE_GOT_MSGID2:
720 rxmsg->msgid |= ((uint32_t)c)<<16;
721 mavlink_update_checksum(rxmsg, c);
722 if(rxmsg->len > 0){
723 status->parse_state = MAVLINK_PARSE_STATE_GOT_MSGID3;
724 } else {
725 status->parse_state = MAVLINK_PARSE_STATE_GOT_PAYLOAD;
726 }
727#ifdef MAVLINK_CHECK_MESSAGE_LENGTH
728 if (rxmsg->len < mavlink_min_message_length(rxmsg) ||
729 rxmsg->len > mavlink_max_message_length(rxmsg))
730 {
731 _mav_parse_error(status);
732 status->parse_state = MAVLINK_PARSE_STATE_IDLE;
733 break;
734 }
735#endif
736 break;
737
738 case MAVLINK_PARSE_STATE_GOT_MSGID3:
739 _MAV_PAYLOAD_NON_CONST(rxmsg)[status->packet_idx++] = (char)c;
740 mavlink_update_checksum(rxmsg, c);
741 if (status->packet_idx == rxmsg->len)
742 {
743 status->parse_state = MAVLINK_PARSE_STATE_GOT_PAYLOAD;
744 }
745 break;
746
747 case MAVLINK_PARSE_STATE_GOT_PAYLOAD: {
748 const mavlink_msg_entry_t *e = mavlink_get_msg_entry(rxmsg->msgid);
749 if (e == NULL) {
750 // Message not found in CRC_EXTRA table.
751 status->parse_state = MAVLINK_PARSE_STATE_GOT_BAD_CRC1;
752 rxmsg->ck[0] = c;
753 } else {
754 uint8_t crc_extra = e->crc_extra;
755 mavlink_update_checksum(rxmsg, crc_extra);
756 if (c != (rxmsg->checksum & 0xFF)) {
757 status->parse_state = MAVLINK_PARSE_STATE_GOT_BAD_CRC1;
758 } else {
759 status->parse_state = MAVLINK_PARSE_STATE_GOT_CRC1;
760 }
761 rxmsg->ck[0] = c;
762
763 // zero-fill the packet to cope with short incoming packets
764 if (e && status->packet_idx < e->max_msg_len) {
765 memset(&_MAV_PAYLOAD_NON_CONST(rxmsg)[status->packet_idx], 0, e->max_msg_len - status->packet_idx);
766 }
767 }
768 break;
769 }
770
771 case MAVLINK_PARSE_STATE_GOT_CRC1:
772 case MAVLINK_PARSE_STATE_GOT_BAD_CRC1:
773 if (status->parse_state == MAVLINK_PARSE_STATE_GOT_BAD_CRC1 || c != (rxmsg->checksum >> 8)) {
774 // got a bad CRC message
775 status->msg_received = MAVLINK_FRAMING_BAD_CRC;
776 } else {
777 // Successfully got message
778 status->msg_received = MAVLINK_FRAMING_OK;
779 }
780 rxmsg->ck[1] = c;
781
782 if (rxmsg->incompat_flags & MAVLINK_IFLAG_SIGNED) {
783 if (status->msg_received == MAVLINK_FRAMING_BAD_CRC) {
784 // If the CRC is already wrong, don't overwrite msg_received,
785 // otherwise we can end up with garbage flagged as valid.
786 status->parse_state = MAVLINK_PARSE_STATE_SIGNATURE_WAIT_BAD_CRC;
787 } else {
788 status->parse_state = MAVLINK_PARSE_STATE_SIGNATURE_WAIT;
789 status->msg_received = MAVLINK_FRAMING_INCOMPLETE;
790 }
791 status->signature_wait = MAVLINK_SIGNATURE_BLOCK_LEN;
792 } else {
793 if (status->signing &&
794 (status->signing->accept_unsigned_callback == NULL ||
795 !status->signing->accept_unsigned_callback(status, rxmsg->msgid))) {
796
797 // If the CRC is already wrong, don't overwrite msg_received.
798 if (status->msg_received != MAVLINK_FRAMING_BAD_CRC) {
799 status->msg_received = MAVLINK_FRAMING_BAD_SIGNATURE;
800 }
801 }
802 status->parse_state = MAVLINK_PARSE_STATE_IDLE;
803 if (r_message != NULL) {
804 memcpy(r_message, rxmsg, sizeof(mavlink_message_t));
805 }
806 }
807 break;
808 case MAVLINK_PARSE_STATE_SIGNATURE_WAIT:
809 case MAVLINK_PARSE_STATE_SIGNATURE_WAIT_BAD_CRC:
810 rxmsg->signature[MAVLINK_SIGNATURE_BLOCK_LEN-status->signature_wait] = c;
811 status->signature_wait--;
812 if (status->signature_wait == 0) {
813 // we have the whole signature, check it is OK
814#ifndef MAVLINK_NO_SIGNATURE_CHECK
815 bool sig_ok = mavlink_signature_check(status->signing, status->signing_streams, rxmsg);
816#else
817 bool sig_ok = true;
818#endif
819 if (!sig_ok &&
820 (status->signing->accept_unsigned_callback &&
821 status->signing->accept_unsigned_callback(status, rxmsg->msgid))) {
822 // accepted via application level override
823 sig_ok = true;
824 }
825 if (status->parse_state == MAVLINK_PARSE_STATE_SIGNATURE_WAIT_BAD_CRC) {
826 status->msg_received = MAVLINK_FRAMING_BAD_CRC;
827 } else if (sig_ok) {
828 status->msg_received = MAVLINK_FRAMING_OK;
829 } else {
830 status->msg_received = MAVLINK_FRAMING_BAD_SIGNATURE;
831 }
832 status->parse_state = MAVLINK_PARSE_STATE_IDLE;
833 if (r_message !=NULL) {
834 memcpy(r_message, rxmsg, sizeof(mavlink_message_t));
835 }
836 }
837 break;
838 }
839
840 // If a message has been successfully decoded, check index
841 if (status->msg_received == MAVLINK_FRAMING_OK)
842 {
843 //while(status->current_seq != rxmsg->seq)
844 //{
845 // status->packet_rx_drop_count++;
846 // status->current_seq++;
847 //}
848 status->current_rx_seq = rxmsg->seq;
849 // Initial condition: If no packet has been received so far, drop count is undefined
850 if (status->packet_rx_success_count == 0) status->packet_rx_drop_count = 0;
851 // Count this packet as received
852 status->packet_rx_success_count++;
853 }
854
855 if (r_message != NULL) {
856 r_message->len = rxmsg->len; // Provide visibility on how far we are into current msg
857 }
858 if (r_mavlink_status != NULL) {
859 r_mavlink_status->parse_state = status->parse_state;
860 r_mavlink_status->packet_idx = status->packet_idx;
861 r_mavlink_status->current_rx_seq = status->current_rx_seq+1;
862 r_mavlink_status->packet_rx_success_count = status->packet_rx_success_count;
863 r_mavlink_status->packet_rx_drop_count = status->parse_error;
864 r_mavlink_status->flags = status->flags;
865 }
866 status->parse_error = 0;
867
868 if (status->msg_received == MAVLINK_FRAMING_BAD_CRC) {
869 /*
870 the CRC came out wrong. We now need to overwrite the
871 msg CRC with the one on the wire so that if the
872 caller decides to forward the message anyway that
873 mavlink_msg_to_send_buffer() won't overwrite the
874 checksum
875 */
876 if (r_message != NULL) {
877 r_message->checksum = rxmsg->ck[0] | (rxmsg->ck[1]<<8);
878 }
879 }
880
881 return status->msg_received;
882}

1001 *
1002 * @endcode

1005{

1007 if (msg_received == MAVLINK_FRAMING_BAD_CRC ||
1008 msg_received == MAVLINK_FRAMING_BAD_SIGNATURE) {

1010 mavlink_message_t* rxmsg = mavlink_get_channel_buffer(chan);
1011 mavlink_status_t* status = mavlink_get_channel_status(chan);
1012 _mav_parse_error(status);

1026/**
1027 * @brief Put a bitfield of length 1-32 bit into the buffer
1028 *
1029 * @param b the value to add, will be encoded in the bitfield
1030 * @param bits number of bits to use to encode b, e.g. 1 for boolean, 2, 3, etc.
1031 * @param packet_index the position in the packet (the index of the first byte to use)
1032 * @param bit_index the position in the byte (the index of the first bit to use)
1033 * @param buffer packet buffer to write into
1034 * @return new position of the last used byte in the buffer
1035 */
1036MAVLINK_HELPER uint8_t put_bitfield_n_by_index(int32_t b, uint8_t bits, uint8_t packet_index, uint8_t bit_index, uint8_t* r_bit_index, uint8_t* buffer)
1037{
1038 uint16_t bits_remain = bits;
1039 // Transform number into network order
1040 int32_t v;
1041 uint8_t i_bit_index, i_byte_index, curr_bits_n;
1042#if MAVLINK_NEED_BYTE_SWAP
1043 union {
1044 int32_t i;
1045 uint8_t b[4];
1046 } bin, bout;
1047 bin.i = b;
1048 bout.b[0] = bin.b[3];
1049 bout.b[1] = bin.b[2];
1050 bout.b[2] = bin.b[1];
1051 bout.b[3] = bin.b[0];
1052 v = bout.i;
1053#else
1054 v = b;
1055#endif
1056
1057 // buffer in
1058 // 01100000 01000000 00000000 11110001
1059 // buffer out
1060 // 11110001 00000000 01000000 01100000
1061
1062 // Existing partly filled byte (four free slots)
1063 // 0111xxxx
1064
1065 // Mask n free bits
1066 // 00001111 = 2^0 + 2^1 + 2^2 + 2^3 = 2^n - 1
1067 // = ((uint32_t)(1 << n)) - 1; // = 2^n - 1
1068
1069 // Shift n bits into the right position
1070 // out = in >> n;
1071
1072 // Mask and shift bytes
1073 i_bit_index = bit_index;
1074 i_byte_index = packet_index;
1075 if (bit_index > 0)
1076 {
1077 // If bits were available at start, they were available
1078 // in the byte before the current index
1079 i_byte_index--;
1080 }
1081
1082 // While bits have not been packed yet
1083 while (bits_remain > 0)
1084 {
1085 // Bits still have to be packed
1086 // there can be more than 8 bits, so
1087 // we might have to pack them into more than one byte
1088
1089 // First pack everything we can into the current 'open' byte
1090 //curr_bits_n = bits_remain << 3; // Equals bits_remain mod 8
1091 //FIXME
1092 if (bits_remain <= (uint8_t)(8 - i_bit_index))
1093 {
1094 // Enough space
1095 curr_bits_n = (uint8_t)bits_remain;
1096 }
1097 else
1098 {
1099 curr_bits_n = (8 - i_bit_index);
1100 }
1101
1102 // Pack these n bits into the current byte
1103 // Mask out whatever was at that position with ones (xxx11111)
1104 buffer[i_byte_index] &= (0xFF >> (8 - curr_bits_n));
1105 // Put content to this position, by masking out the non-used part
1106 buffer[i_byte_index] |= ((0x00 << curr_bits_n) & v);
1107
1108 // Increment the bit index
1109 i_bit_index += curr_bits_n;
1110
1111 // Now proceed to the next byte, if necessary
1112 bits_remain -= curr_bits_n;
1113 if (bits_remain > 0)
1114 {
1115 // Offer another 8 bits / one byte
1116 i_byte_index++;
1117 i_bit_index = 0;
1118 }
1119 }
1120
1121 *r_bit_index = i_bit_index;
1122 // If a partly filled byte is present, mark this as consumed
1123 if (i_bit_index != 7) i_byte_index++;
1124 return i_byte_index - packet_index;
1125}
