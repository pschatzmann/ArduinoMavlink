27/**
28 * @class MavlinkLoggerClass
29 * @brief Simple logger with log levels and configurable Print output (default
30 * Serial).
31 *
32 * Usage:
33 * MavlinkLogger logger;
34 * logger.log(MavlinkLogLevel::INFO, "Hello %d", 42);
35 * logger.setOutput(&myPrint);
36 */
37class MavlinkLoggerClass {
38 public:
39
40 MavlinkLoggerClass() = default;
41
42 /// Set the Print output target (e.g., Serial)
43 void setOutput(Print* out) { output = out; }
44
45 /// Set the minimum log level
46 void setLevel(MavlinkLogLevel lvl) { level = lvl; }
47
48 /// Initialize logger with log level and output (default: INFO, Serial)
49 bool begin(MavlinkLogLevel lvl = INFO, Print* out = &Serial) {
50 setLevel(lvl);
51 setOutput(out);
52 return output != nullptr;
53 }
54
55
56 /// Log an error message (varargs, printf-style)
57 size_t error(const char* fmt, ...) {
58 va_list args;
59 va_start(args, fmt);
60 size_t res = log_va(ERROR, fmt, args);
61 va_end(args);
62 return res;
63 }

70 size_t res = log_va(WARN, fmt, args);
71 va_end(args);
72 return res;
73 }
74
75 /// Log an info message (varargs, printf-style)
76 size_t info(const char* fmt, ...) {
77 va_list args;
78 va_start(args, fmt);
79 size_t res = log_va(INFO, fmt, args);
80 va_end(args);
81 return res;
82 }
83
84
85 /// Log a debug message (varargs, printf-style)
86 size_t debug(const char* fmt, ...) {
87 va_list args;
88 va_start(args, fmt);
89 size_t res = log_va(DEBUG, fmt, args);
90 va_end(args);
91 return res;
92 }
93
94 /// Log a message with a specific log level (varargs, printf-style)
95 size_t log(MavlinkLogLevel lvl, const char* fmt, ...) {
96 va_list args;
97 va_start(args, fmt);
98 size_t res = log_va(lvl, fmt, args);
99 va_end(args);
100 return res;
101 }
102
103 private:
104 Print* output = &Serial;
105 Level level = INFO;
106
107 // Helper for va_list logging
108 size_t log_va(MavlinkLogLevel lvl, const char* fmt, va_list args) {
109 if (lvl > level || !output) return 0;
110 char buf[MAVLINK_MAX_LOG_SIZE];
111 vsnprintf(buf, sizeof(buf), fmt, args);
112 return output->println(buf);
113 }
114};
