15/**
16 * @class MavlinkController
17 * @brief A simple MAVLink drone controller for Arduino/ESP32.
18 *
19 * This class implements a minimal MAVLink drone interface, handling message
20 * reception, parameter storage, and basic telemetry/command processing. It is
21 * based on Dan Zimmerman's barebones_MAVLink example and provides hooks for
22 * custom parameter storage and logging.
23 *
24 * Features:
25 * - Receives and parses MAVLink messages from a serial stream
26 * - Handles basic MAVLink commands (manual control, parameter requests, mission
27 * requests, etc.)
28 * - Periodically transmits heartbeat and telemetry messages
29 * - Supports a user-defined callback for custom message handling before default
30 * processing
31 * - Provides a simple parameter store for application values
32 * - Logging support via user-supplied stream
33 *
34 * Usage:
35 * 1. Instantiate with a Stream and optional ParameterStore
36 * 2. Call loop() regularly (e.g., in Arduino loop())
37 * 3. Optionally set a message callback with setMessageCallback()
38 *
39 * @note This class is designed for embedded use and avoids blocking operations.
40 */
41

84 /// Main loop: processes MAVLink messages and telemetry
85 virtual void loop() {
86 timestamp = millis();
87 while (stream && stream->available() > 0) {
88 int rx = stream->read();
89 if (rx >= 0) {
90 uint8_t rxbyte = rx;
91 mvl_packet_received = mavlink_parse_char(
92 mvl_chan, rxbyte, &mvl_rx_message, &mvl_rx_status);
93 // MAV_DEBUG("available: %d, read byte: 0x%02X, packet received: %d",
94 // stream->available(), rxbyte, mvl_packet_received);
95 if (mvl_packet_received) {
96 MAV_INFO("Received msgid: %d from sysid: %d", mvl_rx_message.msgid,
97 mvl_rx_message.sysid);
98 break;
99 }
100 }
101 }
102 if ((mvl_packet_received) && (mavlink_system_id == mvl_rx_message.sysid)) {
103 bool handled = false;
104 if (messageCallback) {
105 handled = messageCallback(*this, mvl_rx_message);
106 }
107 if (!handled) {
108 processRequest();
109 }
110 }
111 if ((millis() - t_last_hb) > hb_interval) {
112 processHeartbeat();
113 }
114 if ((millis() - t_last_sys_stat) > sys_stat_interval) {
115 processStat();
116 }
117 // feed the dog
118 delay(1);
119 }

171 /// Transmit a MAVLink message, returns bytes written or 0 on error
172 virtual size_t transmitMessage(mavlink_message_t* mvl_msg_ptr) {
173 if (!mvl_msg_ptr) {
174 MAV_ERROR("transmitMessage: null message pointer");
175 return 0;
176 }
177 if (!stream) {
178 MAV_ERROR("transmitMessage: stream is null");
179 return 0;
180 }
181 uint8_t tx_byte_buffer[TX_BUFFER_SIZE] = {0};
182 uint16_t tx_buflen =
183 mavlink_msg_to_send_buffer(tx_byte_buffer, mvl_msg_ptr);
184 if (tx_buflen > TX_BUFFER_SIZE) {
185 MAV_ERROR("transmitMessage: buffer overflow (%d)", tx_buflen);
186 return 0;
187 }
188 return stream->write(tx_byte_buffer, tx_buflen);
189 }
190
191 /// trasmit a command acknowledgment message
192 void transmitAck(uint16_t command, bool accepted) {
193 mavlink_command_ack_t ack{};
194 ack.command = command;
195 ack.result = accepted ? MAV_RESULT_ACCEPTED : MAV_RESULT_UNSUPPORTED;
196 mavlink_msg_command_ack_encode_chan(mvl_sysid, mvl_compid, mvl_chan,
197 &mvl_tx_message, &ack);
198 transmitMessage(&mvl_tx_message);
199 }

201 virtual void handleManualControl(mavlink_message_t* mvl_msg_ptr) {
202 MAV_INFO("handleManualControl");
203 mavlink_manual_control_t mvl_joy;
204 mavlink_msg_manual_control_decode(mvl_msg_ptr, &mvl_joy);
205 setValue(THROTTLE, 0.1f * mvl_joy.z);
206 setValue(YAW, 0.1f * mvl_joy.r);
207 setValue(PITCH, 0.1f * mvl_joy.x);
208 setValue(ROLL, 0.1f * mvl_joy.y);
209 mavlink_msg_manual_control_encode_chan(mvl_sysid, mvl_compid, mvl_chan,
210 &mvl_tx_message, &mvl_joy);
211 transmitMessage(&mvl_tx_message);
212 }

224 virtual void handleCommandLong(mavlink_message_t* mvl_msg_ptr) {
225 MAV_INFO("handleCommandLong");
226 mavlink_command_long_t mvl_cmd;
227 mavlink_msg_command_long_decode(mvl_msg_ptr, &mvl_cmd);
228 switch (mvl_cmd.command) {
229 case (MAV_CMD_REQUEST_AUTOPILOT_CAPABILITIES): {
230 if (1 == mvl_cmd.param1) {
231 mavlink_autopilot_version_t mvl_apv;
232 mvl_apv.flight_sw_version = 2;
233 mvl_apv.middleware_sw_version = 1;
234 mvl_apv.board_version = 1;
235 mvl_apv.vendor_id = 10101;
236 mvl_apv.product_id = 20202;
237 mvl_apv.uid = 0;
238 mvl_apv.capabilities = 0;
239 mvl_apv.capabilities |= MAV_PROTOCOL_CAPABILITY_SET_ATTITUDE_TARGET;
240 mvl_apv.capabilities |= MAV_PROTOCOL_CAPABILITY_MAVLINK2;
241 mavlink_msg_autopilot_version_encode_chan(
242 mvl_sysid, mvl_compid, mvl_chan, &mvl_tx_message, &mvl_apv);
243 transmitMessage(&mvl_tx_message);
244 }
245 break;
246 }
247 case (MAV_CMD_COMPONENT_ARM_DISARM): {
248 mvl_armed = (1 == mvl_cmd.param1) ? 1 : 0;
249 mavlink_command_ack_t mvl_ack;
250 mvl_ack.command = MAV_CMD_COMPONENT_ARM_DISARM;
251 mvl_ack.result = MAV_RESULT_ACCEPTED;
252 mavlink_msg_command_ack_encode_chan(mvl_sysid, mvl_compid, mvl_chan,
253 &mvl_tx_message, &mvl_ack);
254 transmitMessage(&mvl_tx_message);
255 break;
256 }
257 case MAV_CMD_REQUEST_MESSAGE: {
258 if (mvl_cmd.param1 == 259) {
259 transmitAck(MAV_CMD_REQUEST_CAMERA_INFORMATION, camera_active);
260 } else {
261 MAV_INFO("unprocessed MAV_CMD_REQUEST_MESSAGE with param1: %u",
262 mvl_cmd.param1);
263 }
264 break;
265 }
266 case MAV_CMD_REQUEST_CAMERA_INFORMATION: {
267 transmitAck(MAV_CMD_REQUEST_CAMERA_INFORMATION, camera_active);
268 break;
269 }
270 default: {
271 MAV_INFO("Unprocessed command: %d", mvl_cmd.command);
272 }
273 }
274 }

276 virtual void handleMissionRequestList(mavlink_message_t* mvl_msg_ptr) {
277 MAV_INFO("handleMissionRequestList");
278 mavlink_mission_count_t mvl_mc;
279 mvl_mc.target_system = mvl_sysid;
280 mvl_mc.target_component = mvl_compid;
281 mvl_mc.count = 0;
282 mavlink_msg_mission_count_encode_chan(mvl_sysid, mvl_compid, mvl_chan,
283 &mvl_tx_message, &mvl_mc);
284 transmitMessage(&mvl_tx_message);

287 virtual void processRequest() {
288 MAV_INFO("processRequest: %d", mvl_rx_message.msgid);
289 mvl_packet_received = 0;
290 switch (mvl_rx_message.msgid) {
291 case MAVLINK_MSG_ID_HEARTBEAT:
292 // Heartbeat is handled in loop() to ensure regular processing, so
293 // ignore it here
294 MAV_INFO("Received heartbeat");
295 break;
296 case MAVLINK_MSG_ID_MANUAL_CONTROL: {
297 handleManualControl(&mvl_rx_message);
298 break;
299 }
300 case MAVLINK_MSG_ID_PARAM_REQUEST_LIST: {
301 handleParamRequestList(&mvl_rx_message);
302 break;
303 }
304 case MAVLINK_MSG_ID_COMMAND_LONG: {
305 handleCommandLong(&mvl_rx_message);
306 break;
307 }
308 case MAVLINK_MSG_ID_MISSION_REQUEST_LIST: {
309 handleMissionRequestList(&mvl_rx_message);
310 break;
311 }
312 default:
313 MAV_INFO("Unprocessed msgid: %d", mvl_rx_message.msgid);
314 }
315 }
316
317 virtual void processHeartbeat() {
318 MAV_DEBUG("processHeartbeat");
319 mavlink_heartbeat_t mvl_hb;
320 mvl_hb.type = MAV_TYPE_GENERIC;
321 mvl_hb.autopilot = MAV_AUTOPILOT_GENERIC;
322 mvl_hb.system_status = MAV_STATE_ACTIVE;
323 if (mvl_armed) {
324 mvl_hb.base_mode = MAV_MODE_MANUAL_ARMED;
325 mvl_hb.base_mode |= MAV_MODE_FLAG_SAFETY_ARMED;
326 } else {
327 mvl_hb.base_mode = MAV_MODE_MANUAL_DISARMED;
328 }
329 mvl_hb.base_mode |= MAV_MODE_FLAG_MANUAL_INPUT_ENABLED;
330 // Capabilities are not part of the heartbeat message struct. Only set
331 // mavlink_version and other valid fields. Set MAVLink version (2 for v2.0+,
332 // 3 for v2.2+)
333#if defined(MAVLINK_VERSION)
334 mvl_hb.mavlink_version = MAVLINK_VERSION;
335#else
336 mvl_hb.mavlink_version = 1;
337#endif
338 mavlink_msg_heartbeat_encode_chan(mvl_sysid, mvl_compid, mvl_chan,
339 &mvl_tx_message, &mvl_hb);
340 transmitMessage(&mvl_tx_message);
341 t_last_hb = millis();
342 }

344 virtual void processStat() {
345 MAV_DEBUG("processStat");
346 processStatSystem();
347 processStatGPS();
348 processStatRadioStatus();
349 t_last_sys_stat = millis();

352 virtual void processStatSystem() {
353 MAV_DEBUG("processStatSystem");
354
355 mavlink_sys_status_t mvl_sys_stat{};
356 mvl_sys_stat.onboard_control_sensors_present = 0;
357 mvl_sys_stat.onboard_control_sensors_enabled = 0;
358 mvl_sys_stat.onboard_control_sensors_health = 0;
359 mvl_sys_stat.onboard_control_sensors_present_extended = 0;
360 mvl_sys_stat.onboard_control_sensors_enabled_extended = 0;
361 mvl_sys_stat.onboard_control_sensors_health_extended = 0;
362
363 mvl_sys_stat.load = 0;
364 mvl_sys_stat.voltage_battery = getValue(VOLTAGE_BATTERY) * 1000;
365 mvl_sys_stat.current_battery = -1;
366 mvl_sys_stat.battery_remaining = -1;
367 mvl_sys_stat.drop_rate_comm = 0;
368 mvl_sys_stat.errors_comm = 0;
369 mvl_sys_stat.errors_count1 = 0;
370 mvl_sys_stat.errors_count2 = 0;
371 mvl_sys_stat.errors_count3 = 0;
372 mvl_sys_stat.errors_count4 = 0;
373 mavlink_msg_sys_status_encode_chan(mvl_sysid, mvl_compid, mvl_chan,
374 &mvl_tx_message, &mvl_sys_stat);
375 transmitMessage(&mvl_tx_message);
376 }
377
378 virtual void processStatGPS() {
379 MAV_DEBUG("processStatGPS");
380
381 mavlink_global_position_int_t pos;
382 pos.lat = getValue(GPS_LATITUDE) * 1E7;
383 pos.lon = getValue(GPS_LONGITUDE) * 1E7;
384 if (pos.lat != 0.0 || pos.lon != 0.0) {
385 pos.alt = getValue(GPS_ALTITUDE) * 1000;
386 pos.time_boot_ms = timestamp;
387 pos.hdg = getValue(HEADING);
388 mavlink_msg_global_position_int_encode(mvl_sysid, mvl_compid,
389 &mvl_tx_message, &pos);
390 transmitMessage(&mvl_tx_message);
391 }
392 }
393
394 virtual void processStatRadioStatus() {
395 MAV_DEBUG("processStatRadioStatus");
396 mavlink_radio_status_t radio;
397 radio.remrssi = getValue(RECEIVER_RSSI);
398 radio.rssi = getValue(RECEIVER_RSSI);
399 if (radio.remrssi != 0) {
400 mavlink_msg_radio_status_encode(mvl_sysid, mvl_compid, &mvl_tx_message,
401 &radio);
402 transmitMessage(&mvl_tx_message);
403 }
404 }
