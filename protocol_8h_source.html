93static inline uint16_t mavlink_msg_get_send_buffer_length(const mavlink_message_t* msg)
94{
95 if (msg->magic == MAVLINK_STX_MAVLINK1) {
96 return msg->len + MAVLINK_CORE_HEADER_MAVLINK1_LEN+1 + 2;
97 }
98 uint16_t signature_len = (msg->incompat_flags & MAVLINK_IFLAG_SIGNED)?MAVLINK_SIGNATURE_BLOCK_LEN:0;
99 return msg->len + MAVLINK_NUM_NON_PAYLOAD_BYTES + signature_len;
100}

178/*
179 like memcpy(), but if src is NULL, do a memset to zero
180*/
181static inline void mav_array_memcpy(void *dest, const void *src, size_t n)
182{
183 // It would be tempting to do a strcpy/strncpy for the char[] type. Unfortunately, some
184 // existing MAVLink messages such as PARAM_EXT_VALUE.param_value use the char[] type for
185 // arbitrary data (including null), and would break.
186 //
187 // It would be nice to change such char[] types to uint8_t[] but that would change the
188 // CRC_EXTRA.
189
190 if (src == NULL) {
191 memset(dest, 0, n);
192 } else {
193 memcpy(dest, src, n);
194 }
195}
