9/**
10 * @class SimpleUDP
11 * @brief Optional extension of WiFiUDP that ensures MAVLink-compatible Stream
12 * behavior.
13 *
14 * This class extends WiFiUDP to provide the expected Stream interface for
15 * MAVLink and MavlinkController. It ensures that:
16 * - available() returns the size of the next UDP packet if the current one is
17 * consumed
18 * - write() automatically wraps data in beginPacket()/endPacket() for each send
19 *
20 * Usage:
21 * Use SimpleUDP as a drop-in replacement for WiFiUDP when working with
22 * MAVLink or MavlinkController to ensure reliable packet handling and
23 * transmission.
24 */
25class SimpleUDP : public WiFiUDP {
26 public:
27 int available() override {
28 int size = WiFiUDP::available();
29 if (size == 0) {
30 size = parsePacket();
31 }
32 MAV_DEBUG("SimpleUDP::available() = %d", size);
33 return size;
34 }
35 size_t write(const uint8_t* buffer, size_t size) override {
36 MAV_DEBUG("SimpleUDP::write(%d bytes)", size);
37 bool sent = false;
38 size_t written = 0;
39 while (!sent) {
40 beginPacket(remoteIP(), remotePort());
41 written = WiFiUDP::write(buffer, size);
42 sent = endPacket();
43 if (!sent) {
44 delay(20); // Small delay before retrying
45 }
46 }
47 return written;
48 }
49};
