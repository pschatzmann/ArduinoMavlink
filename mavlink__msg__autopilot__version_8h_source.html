81 *
82 * @param capabilities Bitmap of capabilities
83 * @param flight_sw_version Firmware version number.
84 The field must be encoded as 4 bytes, where each byte (shown from MSB to LSB) is part of a semantic version: (major) (minor) (patch) (FIRMWARE_VERSION_TYPE).
85
86 * @param middleware_sw_version Middleware version number
87 * @param os_sw_version Operating system version number
88 * @param board_version HW / board version (last 8 bits should be silicon ID, if any). The first 16 bits of this field specify a board type from an enumeration stored at https://github.com/PX4/PX4-Bootloader/blob/master/board_types.txt and with extensive additions at https://github.com/ArduPilot/ardupilot/blob/master/Tools/AP_Bootloader/board_types.txt
89 * @param flight_custom_version Custom version field, commonly the first 8 bytes of the git hash. This is not an unique identifier, but should allow to identify the commit using the main version number even for very large code bases.
90 * @param middleware_custom_version Custom version field, commonly the first 8 bytes of the git hash. This is not an unique identifier, but should allow to identify the commit using the main version number even for very large code bases.
91 * @param os_custom_version Custom version field, commonly the first 8 bytes of the git hash. This is not an unique identifier, but should allow to identify the commit using the main version number even for very large code bases.
92 * @param vendor_id ID of the board vendor
93 * @param product_id ID of the product
94 * @param uid UID if provided by hardware (see uid2)
95 * @param uid2 UID if provided by hardware (supersedes the uid field. If this is non-zero, use this field, otherwise use uid)
96 * @return length of the message in bytes (excluding serial stream start sign)
97 */
98static inline uint16_t mavlink_msg_autopilot_version_pack(uint8_t system_id, uint8_t component_id, mavlink_message_t* msg,
99 uint64_t capabilities, uint32_t flight_sw_version, uint32_t middleware_sw_version, uint32_t os_sw_version, uint32_t board_version, const uint8_t *flight_custom_version, const uint8_t *middleware_custom_version, const uint8_t *os_custom_version, uint16_t vendor_id, uint16_t product_id, uint64_t uid, const uint8_t *uid2)
100{
101#if MAVLINK_NEED_BYTE_SWAP || !MAVLINK_ALIGNED_FIELDS
102 char buf[MAVLINK_MSG_ID_AUTOPILOT_VERSION_LEN];
103 _mav_put_uint64_t(buf, 0, capabilities);
104 _mav_put_uint64_t(buf, 8, uid);
105 _mav_put_uint32_t(buf, 16, flight_sw_version);

107 _mav_put_uint32_t(buf, 24, os_sw_version);
108 _mav_put_uint32_t(buf, 28, board_version);
109 _mav_put_uint16_t(buf, 32, vendor_id);
110 _mav_put_uint16_t(buf, 34, product_id);
111 _mav_put_uint8_t_array(buf, 36, flight_custom_version, 8);
112 _mav_put_uint8_t_array(buf, 44, middleware_custom_version, 8);
113 _mav_put_uint8_t_array(buf, 52, os_custom_version, 8);
114 _mav_put_uint8_t_array(buf, 60, uid2, 18);
115 memcpy(_MAV_PAYLOAD_NON_CONST(msg), buf, MAVLINK_MSG_ID_AUTOPILOT_VERSION_LEN);
116#else
117 mavlink_autopilot_version_t packet;
118 packet.capabilities = capabilities;
119 packet.uid = uid;
120 packet.flight_sw_version = flight_sw_version;
121 packet.middleware_sw_version = middleware_sw_version;
122 packet.os_sw_version = os_sw_version;
123 packet.board_version = board_version;
124 packet.vendor_id = vendor_id;
125 packet.product_id = product_id;
126 mav_array_memcpy(packet.flight_custom_version, flight_custom_version, sizeof(uint8_t)*8);
127 mav_array_memcpy(packet.middleware_custom_version, middleware_custom_version, sizeof(uint8_t)*8);

129 mav_array_memcpy(packet.uid2, uid2, sizeof(uint8_t)*18);
130 memcpy(_MAV_PAYLOAD_NON_CONST(msg), &packet, MAVLINK_MSG_ID_AUTOPILOT_VERSION_LEN);
131#endif
132
133 msg->msgid = MAVLINK_MSG_ID_AUTOPILOT_VERSION;
134 return mavlink_finalize_message(msg, system_id, component_id, MAVLINK_MSG_ID_AUTOPILOT_VERSION_MIN_LEN, MAVLINK_MSG_ID_AUTOPILOT_VERSION_LEN, MAVLINK_MSG_ID_AUTOPILOT_VERSION_CRC);
