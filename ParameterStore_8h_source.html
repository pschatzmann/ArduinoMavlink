22/**
23 * @class ParameterStore
24 * @brief Flexible parameter storage for MAVLink drone applications.
25 *
26 * This class provides a simple, extensible parameter storage mechanism for
27 * drone applications. It manages an array of float values, indexed by parameter
28 * ID, and can be extended or replaced with custom implementations as needed.
29 *
30 * Features:
31 * - Dynamic allocation of parameter storage
32 * - Safe access with bounds checking
33 * - Easily extensible for custom parameter handling
34 *
35 * Usage:
36 * ParameterStore params;
37 * params.setValue(THROTTLE, 0.5f);
38 * float throttle = params.getValue(THROTTLE);
39 */
40class ParameterStore {
41 public:
42 ParameterStore(uint16_t size = PARAMETER_COUNT) {
43 values = new float[size];
44 this->size = size;
45 for (int j = 0; j < size; j++) {
46 values[j] = 0.0;
47 }
48 }
49 ~ParameterStore() { delete[] values; }
50 float getValue(uint16_t id) {
51 //MAV_DEBUG("ParameterMapStore::getValue(%d)", id);
52 return (id < size) ? values[id] : 0.0;
53 }
54 void setValue(uint16_t id, float value) {
55 MAV_DEBUG("ParameterMapStore::setValue(%d,%f)", id, value);
56 if (id < size) {
57 values[id] = value;
58 }
59 }
60
61 private:
62 float* values = nullptr;
63 uint16_t size;
64};
